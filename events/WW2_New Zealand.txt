# UN relationship
country_event = {
	id = nzl.1
	title = nzl.1.t
	desc = nzl.1.d
	picture = GFX_report_event_generic_read_write

	is_triggered_only = yes

	#Agree with the UN
	option = { 
		name = nzl.1.a
		ai_chance = { factor = 25 }
		every_country = {
			limit = {
				is_in_faction_with = NZL
				is_major = yes
			}
			add_opinion_modifier = { target = NZL modifier = medium_increase }
		}
		every_country = {
			limit = {
				is_in_faction_with = NZL
				NOT = {is_major = yes}
			}
			add_opinion_modifier = { target = NZL modifier = medium_decrease }
		}
	}

	#Criticise the Veto
	option = {
		name = nzl.1.b
		ai_chance = { factor = 75 }

		every_country = {
			limit = {
				is_in_faction_with = NZL
				is_major = yes
			}
			add_opinion_modifier = { target = NZL modifier = medium_decrease }
		}
		every_country = {
			limit = {
				is_in_faction_with = NZL
				NOT = {is_major = yes}
			}
			add_opinion_modifier = { target = NZL modifier = medium_increase }
		}
	}

	#Break from the UN
	option = {
		name = nzl.1.c
		ai_chance = { factor = 5 }

		set_rule = { can_create_factions = yes }
		create_faction = "pacific_forum"
		add_political_power = 150
		add_stability = -0.1
	}
}
