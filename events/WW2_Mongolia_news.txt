#1936 purge
news_event = {
	id = MON_news.6
	hidden = yes

	fire_only_once = yes

	trigger = {
		date > 1936.3.21
		date < 1936.3.25
		country_exists = MON
		MON = { AND = { has_country_leader = { name = "Peljidiin Genden" }
			has_government = communism } }

	}

	immediate = {
		MON = { retire_country_leader = yes
			create_country_leader = {
				name = "Anandyn Amar"
				desc = "POLITICS_ANANDYN_AMAR_DESC"
				picture = "gfx/leaders/MON/Portrait_Anandyn_Amar.dds"
				expire = "1939.3.7"
				ideology = marxism
				traits = {
					nationalist_symbol
				}
			}
		}
	}

	option = {
		name = MON_news.6.a
	}
}
