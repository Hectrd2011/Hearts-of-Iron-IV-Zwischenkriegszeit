# Declaration of War - Triggered by SER
#____________________________________________________________
country_event = {
	id = ww1_firstbalkan.4
	title = ww1_firstbalkan.4.t
	desc = ww1_firstbalkan.4.d
	#picture = ww1_firstbalkan_4
	is_triggered_only = yes
	# SER Yes
	option = {
		name = ww1_firstbalkan.4.a
		ai_chance = {
			factor = 100
		}
		# Serbia
		SER = {
			declare_war_on = {
				type = take_state_focus
				target = TUR
				generator = {
					106
					757
					758
				}
			}
		}
		hidden_effect = {
			SER = {
				add_named_threat = {
					threat = 10
					name = "War in the Balkans"
				}
			}
			set_global_flag = kis_firstbalkan_war
			news_event = {
				days = 1
				id = ww1_firstbalkan.100
			}
			44 = {
				set_border_war = no
			}
			SER = {
				add_manpower = 15000
			}
			BUL = {
				add_manpower = 15000
			}
			GRE = {
				add_manpower = 15000
			}
			MTN = {
				add_manpower = 3500
			}
		}
		# Bulgaria
		if = {
			limit = {
				OR = {
					BUL = {
						is_in_faction_with = SER
					}
					BUL = {
						is_in_faction_with = GRE
					}	
					BUL = {
						is_in_faction_with = MTN
					}
				}
			}
			BUL = {
				create_wargoal = {
					type = take_state_focus
					target = TUR
					generator = {
						757
						754
						755
						760
					}
				}
				add_to_war = {
					targeted_alliance = SER
					enemy = TUR
				}
			}
		}
		# Greece
		if = {
			limit = {
				OR = {
					GRE = {
						is_in_faction_with = SER
					}
					GRE = {
						is_in_faction_with = BUL
					}
					GRE = {
						is_in_faction_with = MTN
					}
				}
				# Greece participates in a war with Turkey only if Turkey did not return the occupied states (731, 184, 756, 757, 790) to it before the start of the war
				TUR = {
					owns_state = 731 
				}	
				TUR = {
					owns_state = 184 
				}	
				TUR = {
					owns_state = 756
				}
				TUR = {
					owns_state = 757
				}
				TUR = {
					owns_state = 790
				}
			}
			GRE = {
				create_wargoal = {
					type = take_state_focus
					target = TUR
					generator = {
						731
						756
						184
					}
				}
				add_to_war = {
					targeted_alliance = SER
					enemy = TUR
				}
			}
		}
		# If Turkey returned Greece to its national states before the start of the war, then Greece leaves the Balkan League and does not participate in the war
		if = {
			limit = {
				GRE = {
					owns_state = 731 
				}	
				GRE = {
					owns_state = 184 
				}	
				GRE = {
					owns_state = 756
				}
				GRE = {
					owns_state = 757
				}
				GRE = {
					owns_state = 790
				}				
			}
			SER = {
				remove_from_faction = GRE
			}
		}		
		# Montenegro
		if = {
			limit = {
				OR = {
					MTN = {
						is_in_faction_with = SER
					}
					MTN = {
						is_in_faction_with = BUL
					}	
					MTN = {
						is_in_faction_with = GRE
					}
				}
			}
			MTN = {
				create_wargoal = {
					type = take_state_focus
					target = TUR
					generator = {
						759
					}
				}
				add_to_war = {
					targeted_alliance = SER
					enemy = TUR
				}
			}
		}
	}
}

country_event = {
	id = ww1_firstbalkan.6
	title = ww1_firstbalkan.6.t
	desc = ww1_firstbalkan.6.d
	#picture = ww1_firstbalkan_6
	fire_only_once = yes
	is_triggered_only = yes
	# SER Yes
	option = {
		name = ww1_firstbalkan.6.a
		ai_chance = {
			factor = 100
		}
		# If Greece did not participate in the First Balkan War, then region 758 (Central Epirus) will not go to it, but to Albania
		if = {
			limit = {
				NOT = {
					GRE = {
						has_war_with = TUR
					}
				}
			}	
			ALB = {
				transfer_state = 758
				add_state_core = 758
			}		
		}	
		# If Greece did not participate in the First Balkan War, then region 731 (West Macedonia) will not go to it, but to Bulgaria
		# However, if region 731 (West Macedonia) already belongs to Greece (it was transferred by Turkey before the start of the war), then it remains with Greece		
		if = {
			limit = {
				NOT = {
					GRE = {
						has_war_with = TUR
					}
					GRE = {
						owns_state = 731 
					}
				}
			}	
			BUL = {
				transfer_state = 731
				add_state_core = 731
			}			
		}			
		# If Greece participate in the First Balkan War, then region 731 (West Macedonia), 758 (Central Epirus) and 790 (North Aegean islands) will go to it
		if = {
			limit = {
				GRE = {
					has_war_with = TUR
				}
			}	
			GRE = {
				white_peace = TUR
				transfer_state = 731
				transfer_state = 758
				transfer_state = 790
				add_state_core = 731
				add_state_core = 758
				add_state_core = 790
				diplomatic_relation = {
					country = TUR
					relation = non_aggression_pact
				}
			}	
		}	
		# If Bulgaria did not participate in the First Balkan War, then regions 184 (East Macedonia) and 759 (Nordeastern Macedonia) will not go to it, but to Greece and Serbia
		if = {
			limit = {
				NOT = {
					BUL = {
						has_war_with = TUR
					}
				}	
			}		
			GRE = {
				transfer_state = 184
				add_state_core = 184
			}	
			SER = {
				transfer_state = 759
				add_state_core = 759
			}			
		}	
		# If Bulgaria participate in the First Balkan War, then regions 788 (East Rumelia) and 759 (Nordeastern Macedonia) will go to it
		if = {
			limit = {
				BUL = {
					has_war_with = TUR
				}
			}		
			BUL = {
				white_peace = TUR
				transfer_state = 788
				transfer_state = 759
				add_state_core = 788
				add_state_core = 759
				diplomatic_relation = {
					country = TUR
					relation = non_aggression_pact
				}
			}	
		}	
		# If Turkey returned Greece to its national states (184, 756, 757) before the start of the war, then it remains with Greece and not transferred to Bulgaria
		if = {
			limit = {
				TUR = {
					owns_state = 184 
				}	
				TUR = {
					owns_state = 756
				}
				TUR = {
					owns_state = 757
				}	
			}		
			BUL = {
				transfer_state = 184
				transfer_state = 756
				transfer_state = 757
				add_state_core = 184
				add_state_core = 756
				add_state_core = 757
			}	
		}	

		MTN = {
			white_peace = TUR
		}		
		MTN = {
			transfer_state = 761
			add_state_core = 761
			diplomatic_relation = {
				country = TUR
				relation = non_aggression_pact
			}
		}
		ALB = {
			transfer_state = 44
			transfer_state = 803
			add_state_core = 44
			add_state_core = 803
			diplomatic_relation = {
				country = TUR
				relation = non_aggression_pact
			}
		}
		SER = {
			white_peace = TUR
		}		
		SER = {
			transfer_state = 106
			transfer_state = 760
			add_state_core = 106
			add_state_core = 760
			diplomatic_relation = {
				country = TUR
				relation = non_aggression_pact
			}
			remove_from_faction = ITA
		}		
		hidden_effect = {
			set_global_flag = kis_firstbalkanwar_victory
			news_event = {
				days = 1
				id = ww1_firstbalkan.101
			}
			# Diplomatics relations changed to avoid AI to do stupid stuff
			GRE = {
				diplomatic_relation = {
					country = TUR
					relation = non_aggression_pact
				}
				add_opinion_modifier = {
					target = TUR
					modifier = HOI4TGW_Opinion_hostile_relations_post_balkan_war
				}
			}
			BUL = {
				diplomatic_relation = {
					country = TUR
					relation = non_aggression_pact
				}
				add_opinion_modifier = {
					target = TUR
					modifier = HOI4TGW_Opinion_hostile_relations_post_balkan_war
				}
			}
			MTN = {
				diplomatic_relation = {
					country = TUR
					relation = non_aggression_pact
				}
				add_opinion_modifier = {
					target = TUR
					modifier = HOI4TGW_Opinion_hostile_relations_post_balkan_war
				}
			}
			ALB = {
				diplomatic_relation = {
					country = TUR
					relation = non_aggression_pact
				}
			}
			SER = {
				diplomatic_relation = {
					country = TUR
					relation = non_aggression_pact
				}
				add_opinion_modifier = {
					target = TUR
					modifier = HOI4TGW_Opinion_hostile_relations_post_balkan_war
				}
			}
			TUR = {
				add_opinion_modifier = {
					target = SER
					modifier = HOI4TGW_Opinion_hostile_relations_post_balkan_war
				}
				add_opinion_modifier = {
					target = BUL
					modifier = HOI4TGW_Opinion_hostile_relations_post_balkan_war
				}
				add_opinion_modifier = {
					target = GRE
					modifier = HOI4TGW_Opinion_hostile_relations_post_balkan_war
				}
				add_opinion_modifier = {
					target = MTN
					modifier = HOI4TGW_Opinion_hostile_relations_post_balkan_war
				}
			}
		}
	}
