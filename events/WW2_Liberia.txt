	desc = britain.700.d

	option = { # Accept purchase
		name = britain.700.a

			factor = 90

						is_puppet_of = GER

					}
				}
				LIB = {
					surrender_progress > 0.09
				}
			}
			modifier = {
				factor = 0

					is_in_faction_with = ROOT
					has_war_with = ENG
				}
			}
			modifier = {
				factor = 0
				any_other_country = {
					NOT = {
						tag = GER
					}
					LIB = {
						is_puppet_of = PREV
					}
					NOT = {
						is_in_faction_with = GER
					}
				}
			}
		}
		LIB = {
			country_event = { days = 1 id = liberia.2 }
		}

						}
					}
				}
			}
			modifier = {
				factor = 90
				OR = {
					is_in_faction_with = GER
					LIB = {
						is_puppet_of = GER
					}
				}
			}
		}
		LIB = {
			country_event = { days = 1 id = liberia.3 }
