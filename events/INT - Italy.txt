add_namespace = INT_italy

# March on Rome
country_event = {
	id = INT_italy.1
	title = INT_italy.1.t
	desc = INT_italy.1.d
	picture = GFX_report_event_march_on_rome

	fire_only_once = yes

	trigger = {
		NOT = { date > 1932.12.1 }
		tag = ITA
		date > 1922.10.21
		NOT = { has_government = fascism }
	}

	mean_time_to_happen = { days = 1 }

	immediate = {
		set_global_flag = march_on_rome
	}
	option = {
		name = INT_italy.1.a
		set_politics = {
			ruling_party = fascism
			elections_allowed = no }

		set_popularities = {
			fascism = 50
			democratic = 35
			communism = 15
		}
		retire_country_leader = yes
		create_country_leader = {
			name = "Benito Mussolini"
			desc = "POLITICS_MUSSO_THE_DUSSO_DESC"
			picture = "Portrait_Italy_Benito_Mussolini.dds"
			expire = "1963.1.1"
			ideology = italian_fascism
			traits = {
				dictator
			}
		}

		hidden_effect = {
			country_event = { id = INT_italy.2 days = 304 }
		}
	}
}
