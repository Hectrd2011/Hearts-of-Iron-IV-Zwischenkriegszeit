###########################
# News Events
###########################

add_namespace = pacnews

# Japan Strike USA
news_event = {
	id = pacnews.1
	title = pacnews.1.t
	desc = pacnews.1.d
	picture = GFX_news_event_japan_strike_usa

	major = yes

	is_triggered_only = yes

	option = {
		name = pacnews.1.a
		trigger = {
			AND = {
				NOT = { has_government = fascism }
				NOT = {
					TAG = JAP
					TAG = USA
				}
			}
		}
	}
	option = {
		name = pacnews.1.b
		trigger = { TAG = JAP }
	}
	option = {
		name = pacnews.1.c
		trigger = { TAG = USA }
	}
	option = {
		name = pacnews.1.e
		trigger = {
			AND = {
				has_government = fascism
				NOT = {
					TAG = JAP
					TAG = USA
				}
			}
		}
	}

}
