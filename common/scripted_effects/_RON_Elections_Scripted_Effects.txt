RON_handle_Argentine_Elections = {
	#1937 Election
	if = {
		limit = {
			date > 1936.1.1
			date < 1938.1.1
		}
		country_event = { id = raelections.5 }
	}
	#1946 Election
	else_if = {
		limit = {
			date > 1945.1.1
			date < 1947.1.1
		}
		country_event = { id = CW_Expansion_country.10 }
	}
	#1951 Election
	else_if = {
		limit = {
			date > 1951.1.1
			date < 1953.1.1
		}
		country_event = { id = CW_Expansion_country.245 }
	}

	else = {
		country_event = { id = election.1 }
	}
}
