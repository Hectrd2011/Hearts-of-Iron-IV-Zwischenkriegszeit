initiate_arabian_war = {
    every_country = {
        limit = {
            is_arabian_tag = yes
        }
        every_country = {
            limit = {
                is_arabian_tag = yes
                NOT = {
                    has_war_with = PREV
                }
            }
            declare_war_on = {
                target = PREV
                type = civil_war
            }
            add_civil_war_target = PREV
        }
    }
    every_country = {
        limit = {
            is_arabian_tag = yes
            NOT = {
                tag = OTT
            }
        }
        division_template = {
            name = "Militia"
            is_locked = yes
   
            regiments = {
                infantry = { x = 0 y = 0 }
                infantry = { x = 0 y = 1 }
                infantry = { x = 0 y = 2 }
                infantry = { x = 0 y = 3 }
            }
        }
        for_each_scope_loop = {
	        array = core_states
            if= {
                limit = {
                    THIS = {
                        is_controlled_by = PREV
                    }
                }
                create_unit = {
                    division = "name = \"Militias\" division_template = \"Militia\" start_experience_factor = 0.5 start_equipment_factor = 0.5 start_manpower_factor = 0.5"
                    owner = PREV
                }
            }
            
	
        }
    }
    
}
