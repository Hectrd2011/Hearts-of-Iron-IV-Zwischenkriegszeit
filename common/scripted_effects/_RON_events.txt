RON_give_russian_winter_event = {
	if = {
		limit = {
			country_exists = SOV
			NOT = { has_idea = russian_winter }
		}
		SOV = {
			country_event = { id = weather_events.1 }
		}
	}
	if = {
		limit = {
			country_exists = RUS
			NOT = { has_idea = russian_winter }
		}
		RUS = {
			country_event = { id = weather_events.1 }
		}
	}
	set_global_flag = RON_weather_report_in_progress
}
